#datetime module
from datetime import datetime #IN THIS CASE< MODULE AND OBJECT HAVE SAME NAME

current_date = datetime.now()
print(current_date) #PRINTS CURRENT DATE AND TIME

mybirthday = datetime(2005, 12, 9, 5, 23, 45)    #this prints any date i want. datetime(year, month, day, hour, second)
print(mybirthday)
print(mybirthday.hour)  #output hour of my birthday. can also do  .year, .day .minute etc


custom_date_difference = datetime(2026, 7, 1) - datetime(2018, 12, 5)
print(custom_date_difference)   #this outputs the difference in the date by days, in this case, 2765 days
print(datetime.now() - datetime(2018, 1, 1))   #can do it with .now function aswell

#strptime extracts a date from a string using certain arguments.  to know which ones to use like below, look it up
extracted_date_from_str = datetime.strptime("January 18 2025", "%B %d %Y")   #syntax: datetime.strptim(string, code code code)  depends on format code changes. the code for "jan" is different for example
print(extracted_date_from_str)
print(extracted_date_from_str.year)  #outputs 2025

#strftime works the opposite way. it turns a datetime into a string to use as a string
string_date = datetime.strftime(datetime.now(), "%d %b %Y")
print(string_date)  #output  "16 jun 2026"
